	desc = mon.8.d

	trigger = {
		tag = MON
		MON = { has_country_flag = chadhat }
		PRC = { exists = yes }
		OR = {
			MON = { NOT = { has_government = communism } }
			AND = {
				MON = { has_war_with = PRC }
				PRC = { has_government = communism }
			}
		}
	}

	mean_time_to_happen = {
		days = 3
	}

	fire_only_once = yes
	
	option = { #
		name = mon.8.a
		ai_chance = { factor = 100 }
		remove_unit_leader = 19237
		effect_tooltip = {
			PRC = {
				create_corps_commander = {
					name = "Ulanhu"

					planning_skill = 5
